220  READ(8,nml=aparams)

224  CLOSE(8)

227 
228  phys%n = n
229  phys%rra = rra

232  phys%kp = kp
233  phys%r = r
234  phys%d = d
235  phys%f0 = f0
236  phys%gp = gp
237  phys%H = h
238  phys%phi0_npi = phi0_npi
239 
240  phys%lambda = lambda
241  phys%Co = co
242  phys%Go = go
243  phys%Ca = ca
244  phys%To0 = to0
245  phys%Ta0 = ta0
246  phys%epsa = epsa
247  phys%Ga = ga

250  phys%scale = scale

252  phys%sc = sc
253  phys%sB = sb
254 
255  phys%nua = nua
256  phys%nuo = nuo
257 
258  ! Some general parameters (Domain, beta, gamma, coupling) !
259  !---------------------------------------------------------!
260 
261  phys%pi=dacos(-1.d0)
262  phys%L=phys%scale/phys%pi
263  phys%phi0=phys%phi0_npi*phys%pi
264  phys%LR=sqrt(phys%gp*phys%H)/phys%f0
265  phys%G=-phys%L**2/phys%LR**2
266  phys%betp=phys%L/phys%rra*cos(phys%phi0)/sin(phys%phi0)
267  phys%rp=phys%r/phys%f0
268  phys%dp=phys%d/phys%f0
269  phys%kd=phys%k*2
270  phys%kdp=phys%kp
271 
272  ! DERIVED QUANTITIES !
273  !-----------------------------------------------------!
274 
275  phys%Cpo=phys%Co/(phys%Go*phys%f0) * phys%RR/(phys%f0**2*phys%L**2)
276  phys%Lpo=phys%lambda/(phys%Go*phys%f0)
277  phys%Cpa=phys%Ca/(phys%Ga*phys%f0) * phys%RR/(phys%f0**2*phys%L**2)/2 ! Cpa acts on psi1-psi3, not on theta
278  phys%Lpa=phys%lambda/(phys%Ga*phys%f0)
279  phys%sBpo=4*phys%sB*phys%To0**3/(phys%Go*phys%f0) ! long wave radiation lost by ocean to atmosphere space
280  phys%sBpa=8*phys%epsa*phys%sB*phys%Ta0**3/(phys%Go*phys%f0) ! long wave radiation from atmosphere absorbed by ocean
281  phys%LSBpo=2*phys%epsa*phys%sB*phys%To0**3/(phys%Ga*phys%f0) ! long wave radiation from ocean absorbed by atmosphere
282  phys%LSBpa=8*phys%epsa*phys%sB*phys%Ta0**3/(phys%Ga*phys%f0) ! long wave radiation lost by atmosphere to space & ocea
283  phys%nuap=phys%nua/(phys%f0*phys%L**2)
284  phys%nuop=phys%nuo/(phys%f0*phys%L**2)
285 
286  phys%initialized = .true.
287

343 
344  mode_selection%oms = oms
345  mode_selection%ams = ams
346 
347  ! Computation of the dimension of the atmospheric !
348  ! and oceanic components !
349  !---------------------------------------------------------!
