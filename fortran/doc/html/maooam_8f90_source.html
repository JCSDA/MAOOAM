23  REAL(KIND=8), DIMENSION(:), ALLOCATABLE :: x !< State variable in the model
24  REAL(KIND=8), DIMENSION(:), ALLOCATABLE :: xnew !< Updated state variable
25  REAL(KIND=8) :: t=0.d0 !< Time variable
26  REAL(KIND=8) :: t_up

33  LOGICAL, POINTER :: writeout
34 
35  print*, 'Model MAOOAM v1.4'
36  print*, 'Loading information...'
37 
38  ! Initializing cont_evol
39  cont_evol=.false.
40  arg='arg'
41  i=0
42  DO WHILE ((.NOT. cont_evol) .AND. len_trim(arg) /= 0)
43  CALL get_command_argument(i, arg)
44  IF (trim(arg)=='continue') cont_evol=.true.
45  i=i+1
46  END DO
47 
48  CALL maooam_model%init
49  CALL integr%init(maooam_model)
50 
51  tw => maooam_model%model_configuration%integration%tw
52  tw_snap => maooam_model%model_configuration%integration%tw_snap
53  t_trans => maooam_model%model_configuration%integration%t_trans
54  t_run => maooam_model%model_configuration%integration%t_run
55  writeout => maooam_model%model_configuration%integration%writeout
56  ndim => maooam_model%model_configuration%modes%ndim
57 
58  t_up=integr%dt/t_trans*100.d0
59 
60  IF (writeout) OPEN(10,file='evol_field.dat')
61 
62  ALLOCATE(x(0:ndim),xnew(0:ndim))
63 
64  x = maooam_model%load_IC()
65 
66  IF (cont_evol) THEN
67  INQUIRE(file='snapshots_trans.dat',exist=ex,nextrec=next)
68  IF (ex) THEN
69  OPEN(11,file='snapshots_trans.dat')
70  READ(11,rec=next-1) x
71  CLOSE(11)
72  END IF
73  END IF
74 
75  IF (writeout) OPEN(11,file='snapshots_trans.dat',status='replace',form='UNFORMATTED',access='DIRECT',recl=8*ndim)
76 
77  print*, 'Starting the transient time evolution...'
78 
79  indexsnap=0
80  DO WHILE (t<t_trans)
81  CALL integr%step(x,t,xnew)
82  x=xnew
83  IF (mod(t/t_trans*100.d0,0.1)<t_up) WRITE(*,'(" Progress ",F6.1," %",A,$)') t/t_trans*100.d0,char(13)
84  IF (writeout .AND. mod(t,tw_snap)<integr%dt) THEN
85  indexsnap=indexsnap+1
86  WRITE(11,rec=indexsnap,iostat=wrstat) x
87  END IF
88  END DO
89 
90  IF (writeout) CLOSE(11)
91

93 
94  CALL stat_acc%init(ndim)
95 
96  t=0.d0
97  t_up=integr%dt/t_run*100.d0
98 
99  IF (writeout) WRITE(10,*) t,x(1:ndim)
100 
101  DO WHILE (t<t_run)
102  CALL integr%step(x,t,xnew)
103  x=xnew
104  IF (mod(t,tw)<integr%dt) THEN
105  IF (writeout) WRITE(10,*) t, x(1:ndim)
106  CALL stat_acc%accumulate(x(1:ndim))
107  END IF
108  IF (mod(t/t_run*100.d0,0.1)<t_up) WRITE(*,'(" Progress ",F6.1," %",A,$)') t/t_run*100.d0,char(13)
109  END DO
110 
111  print*, 'Evolution finished.'
112 
113  IF (writeout) CLOSE(10)
114 
115  IF (writeout) OPEN(10,file='mean_field.dat')
116 
117  IF (writeout) WRITE(10,*) stat_acc%mean()
118  IF (writeout) CLOSE(10)
119 
120  DEALLOCATE(x,xnew)
121  CALL maooam_model%clean
122  CALL integr%clean
123  CALL stat_acc%clean
124 
125 END PROGRAM maooam
